Animation

public class Animation : AnimationBasic

The basic class for single-keyframe animations.

        Animation(type: .color)
            .duration(duration)
            .from(UIColor)
            .to(UIColor)

        Animation(type: .textColor)
            .duration(duration)
            .from(UIColor)
            .to(UIColor)

        Animation(type: .position)
            .duration(duration)
            .from(x, y)
            .to(x, y)
            .resetOnStop(true)

        Animation(type: .rotation)
            .duration(duration)
            .from(fromValue)
            .to(toValue)

        Animation(type: .scale)
            .duration(duration)
            .from(xScale, yScale)
            .to(xScale, yScale)

        Animation(type: .contentOffset)
            .duration(duration)
            .from(x, y)
            .to(x, y)
